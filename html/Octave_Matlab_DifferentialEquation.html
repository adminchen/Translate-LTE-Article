Ex)

Input

f = inline('-2*y','t','y');

odeopt = odeset ("RelTol", 0.00001, "AbsTol", 0.00001,"InitialStep",0.5,"MaxStep",0.5);

[t,y] = ode45(f,[0 5],2,odeopt);

plot(t,y)

[y,t,x] = lsim(sys,u,t,x0);
