< Matlab/Octave Code I >

 

    xmin = -1;

    xmax = 4;

    ymin = -5;

    ymax = 15;

     

    x = xmin:0.1:xmax;

     

    fx = @(x)[x .* (x .- 2) + 5];

     

    x1 = 1.5;

    y1 = fx(x1);

    x2 = 2.5;

    y2 = fx(x2);

     

    a = (y1-y2)/(x1-x2);

    b = (x1*y2-x2*y1)/(x1-x2);

    yd = a .* x + b;

     

    hold on;

    plot(x,fx(x),'k-','Linewidth',2);axis([xmin xmax ymin ymax]);

    plot(x,yd,'r-');axis([xmin xmax ymin ymax]);

    stem([x1 x2],[y1 y2],'MarkerFaceColor',[0 1 0]);

    grid;

    hold off;

 

 

< Matlab/Octave Code II >

 

    xmin = -1;

    xmax = 4;

    ymin = -5;

    ymax = 15;

     

    x = xmin:0.1:xmax;

     

    fx = @(x)[x .* (x .- 2) + 5];

     

    x1 = 2.0;

    y1 = fx(x1);

    hmax = 1.5;

    hstep = hmax/16;

     

    for i = 1:16,

     

      x2 = x1 + hmax - i*hstep + 0.0001;

      y2 = fx(x2);

       

      a = (y1-y2)/(x1-x2);

      b = (x1*y2-x2*y1)/(x1-x2);

      yd = a .* x + b;

       

      subplot(4,4,i);

       

      hold on;

      plot(x,fx(x),'k-','Linewidth',2);axis([xmin xmax ymin ymax]);

      plot(x,yd,'r-');axis([xmin xmax ymin ymax]);

      stem([x1 x2],[y1 y2],'MarkerFaceColor',[0 1 0]);

      grid;

      hold off;

     

    end;

 

 

< Matlab/Octave Code III >

     

    xmin = -2;

    xmax = 2;

    ymin = -2;

    ymax = 2;

     

    x0 = 0.2;

    x1 = 0.20001;

    y0 = -0.2;

    y1 = -0.20001;

     

    x = xmin:0.1:xmax;

    y = ymin:0.1:ymax;

    [X,Y] = meshgrid(x,y);

     

    fxy = @(x,y)[exp(-(x.^2+y.^2))];

    Z = fxy(X,Y);

     

    Zx = exp(-(x.^2+y0.^2));

    Zy = exp(-(x0.^2+y.^2));

     

    zx0 = fxy(x0,y0);

    zx1 = fxy(x1,y0);

     

    a = (zx0-zx1)/(x0-x1);

    b = (x0*zx1-x1*zx0)/(x0-x1);

    Zxd = a .* x + b;

     

    zy0 = fxy(x0,y0);

    zy1 = fxy(x0,y1);

     

    a = (zy0-zy1)/(y0-y1);

    b = (y0*zy1-y1*zy0)/(y0-y1);

    Zyd = a .* y + b;

     

    hold on;

    mesh(X,Y,Z);

    plot3(x,y .* 0 .+ y0, Zx, 'linewidth',2,'color','k');axis([-2 2 -2 2 0 1.2]);

    plot3(x .* 0 .+ x0,y, Zy, 'linewidth',2,'color','k');axis([-2 2 -2 2 0 1.2]);

    plot3(x,y .* 0 .+ y0, Zxd, 'linewidth',2,'color','r');axis([-2 2 -2 2 0 1.2]); % slot at x0,y0 in x direction

    plot3(x .* 0 .+ x0,y, Zyd, 'linewidth',2,'color','r');axis([-2 2 -2 2 0 1.2]); % slot at x0,y0 in y direction

    view(-20,50);

    hold off;
